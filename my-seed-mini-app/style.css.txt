    <script>
        // Initialize Telegram WebApp
        let tg = window.Telegram.WebApp;
        let user = null;

        // When page loads
        window.onload = function() {
            tg.expand();
            tg.ready();
            
            // Get user info from Telegram
            user = tg.initDataUnsafe.user;
            
            if (user) {
                document.getElementById('welcome-text').textContent = 
                    'Welcome, ' + user.first_name + '!';
                
                // Load student data (fake data for now)
                setTimeout(function() {
                    loadStudentData();
                }, 1000);
            } else {
                alert('Please open this app through Telegram');
            }
        };

        function loadStudentData() {
            // Fake data - in real app, this comes from your server
            var studentData = {
                payment_verified: true,
                plan_expiry: '2024-04-20',
                upcoming_classes: [
                    {subject: 'Physics', day: 'Monday', time: '4:00 PM'},
                    {subject: 'Maths', day: 'Wednesday', time: '2:00 PM'}
                ]
            };

            // Update payment status
            var statusEl = document.getElementById('payment-status');
            var planInfoEl = document.getElementById('plan-info');
            
            if (studentData.payment_verified) {
                statusEl.textContent = '✅ Verified';
                statusEl.className = 'status-verified';
                planInfoEl.textContent = 'Plan expires: ' + studentData.plan_expiry;
            } else {
                statusEl.textContent = '❌ Not Verified';
                statusEl.className = 'status-pending';
                planInfoEl.textContent = 'Please verify payment to book classes';
            }

            // Update classes list
            var classesList = document.getElementById('classes-list');
            if (studentData.upcoming_classes.length > 0) {
                var html = '';
                for (var i = 0; i < studentData.upcoming_classes.length; i++) {
                    var c = studentData.upcoming_classes[i];
                    html += '<div style="padding: 10px; background: #f0f2f5; border-radius: 8px; margin-bottom: 8px;">';
                    html += '<strong>' + c.subject + '</strong><br>';
                    html += '<small>' + c.day + ' at ' + c.time + '</small>';
                    html += '</div>';
                }
                classesList.innerHTML = html;
            }

            // Hide loading, show content
            document.getElementById('loading').classList.add('hidden');
            document.getElementById('main-content').classList.remove('hidden');
        }

        function bookClass() {
            // Send message back to bot
            tg.sendData(JSON.stringify({action: 'book_class'}));
            tg.close();
        }

        function viewSchedule() {
            tg.sendData(JSON.stringify({action: 'view_schedule'}));
            tg.close();
        }

        function verifyPayment() {
            tg.sendData(JSON.stringify({action: 'verify_payment'}));
            tg.close();
        }

        function viewProgress() {
            tg.sendData(JSON.stringify({action: 'view_progress'}));
            tg.close();
        }

        function contactTeacher() {
            tg.sendData(JSON.stringify({action: 'contact'}));
            tg.close();
        }
    </script>
